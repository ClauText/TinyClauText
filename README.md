# TinyClauText
    Tiny Version of ClauText
    Just for Integer, +,-,*,/,%
    and support print function, if-else, while statements.

# Goal?
    x = 1 y = 2
    Main = { Event = { id = 1 } }
    Event = {
        id = 1
        
        $print = { { /./x } }
        $print = { { _ } }
        $print = { { /./y } }
        
        $_getch = { }
    }
